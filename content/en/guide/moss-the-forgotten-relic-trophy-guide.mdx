  <Step num="2" title="Play Defensively">

  <Step num="3" title="Prioritize Healing">
    Keep your hands near Quill at all times. If you see her glowing red or panting, heal her immediately. Do not wait for the combat encounter to end.
  </Step>
  <Step num="4" title="Slow Down for Platforming">

  </Step>
</Steps>
